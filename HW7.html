\[ logit(p) = \beta_0 + \beta_1\text{CAR}+\beta_2\text{AGE}+\beta_3\text{DIST} + \epsilon \]

\[ logit(p)=-1.667+0.232(\text{CAR})-.210(\text{AGE})+.259(\text{DIST}) \]
